<!DOCTYPE html>
<html>
<head>
    <title>Day of the Week</title>
</head>
<body>

<script>
// Prompt the user to enter a number from 1 to 7
const userInput = prompt("Enter a number from 1 to 7:");

// Convert the input to a number
const dayNumber = parseInt(userInput);

// Check if the entered value is a valid number between 1 and 7
if (isNaN(dayNumber) || dayNumber < 1 || dayNumber > 7) {
    document.body.innerHTML = "<h2>Invalid input! Please enter a number from 1 to 7.</h2>";
} else {
    // Create an array to map day numbers to day names
    const daysOfWeek = ["Monday", "Tuesday", "Wednesday", "Thursday", "Friday", "Saturday", "Sunday"];

    // Get the day name based on the input number
    const dayName = daysOfWeek[dayNumber - 1];

    // Display the result in the body of the page
    document.body.innerHTML = "<h2> " + dayName + "</h2>";
}
</script>

</body>
</html>
